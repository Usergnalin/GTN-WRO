 22    def line_trace_time(
 23        self,
 24        duration : int,
 25        Kp, Ki, Kd,
 26        mode="balance",
 27        polling_rate : int = 5,
 28        then="HOLD",
 29        TRACE_TARGET=50,):
 30        """
 31        PID line tracing with multiple modes.
 32
 33        Parameters:
 34        - duration: total running time in ms
 35        - Kp, Ki, Kd: PID constants
 36        - mode: string, one of "balance", "left_only", "right_only", "left_minus_right", "right_minus_left"
 37        - polling_rate: wait time between updates in ms
 38        - then: what to do after duration ends: "HOLD", "STOP", or "BRAKE"
 39        """
 40
 41        integral = 0
 42        last_error = 0
 43        start = self.watch.time()  # Start stopwatch
 44
 45        while (self.watch.time() - start) < duration:
 46            left_val = self.left_sensor.reflection()
 47            right_val = self.right_sensor.reflection()
 48
 49            if left_val is not None and right_val is not None:
 50                # Calculate error based on mode
 51                if mode == "balance":
 52                    error = (TRACE_TARGET - left_val) - (TRACE_TARGET - right_val)
 53                elif mode == "left_only":
 54                    error = TRACE_TARGET - left_val
 55                elif mode == "right_only":
 56                    error = TRACE_TARGET - right_val
 57                elif mode == "left_minus_right":
 58                    error = left_val - right_val
 59                elif mode == "right_minus_left":
 60                    error = right_val - left_val
 61                else:
 62                    error = 0  # default no correction
 63
 64                integral += error
 65                derivative = error - last_error
 66
 67                turn = Kp * error + Ki * integral + Kd * derivative
 68
 69                speed_left = min(max(self.TRACE_SPEED + turn, 0), self.MAX_SPEED)
 70                speed_right = min(max(self.TRACE_SPEED - turn, 0), self.MAX_SPEED)
 71
 72                self.left_motor.run(speed_left)
 73                self.right_motor.run(speed_right)
 74
 75                last_error = error
 76
 77            wait(polling_rate)
 78
 79        # After loop ends
 80        if then == "HOLD":
 81            self.left_motor.hold()
 82            self.right_motor.hold()
 83        elif then == "STOP":
 84            self.left_motor.stop()
 85            self.right_motor.stop()
 86        elif then == "BRAKE":
 87            self.left_motor.brake()
 88            self.right_motor.brake()

 90    def turn_arc(self, angle: float, radius_factor=0.0, then="HOLD"):
 91        """
 92        Turns the robot along an arc using differential wheel speeds.
 93
 94        Parameters:
 95        - angle: float, turn amount in degrees. Positive for right, negative for left.
 96        - radius_factor: float, adjusts arc radius (0 = in-place turn, higher = wider arc).
 97        - then: action after turn ends. One of "HOLD", "STOP", or "BRAKE".
 98        """
 99        if then == "HOLD":
100            then = Stop.HOLD
101        elif then == "STOP":
102            then = Stop.COAST
103        elif then == "BRAKE":
104            then = Stop.BRAKE
105
106        left_angle = angle * self.TURN_CONST + radius_factor
107        right_angle = -angle * self.TURN_CONST + radius_factor
108
109        abs_left = abs(left_angle)
110        abs_right = abs(right_angle)
111
112        if abs_left > abs_right:
113            ratio = abs_left / abs_right
114            speed_left = self.BASE_SPEED
115            speed_right = self.BASE_SPEED / ratio
116        else:
117            ratio = abs_right / abs_left
118            speed_right = self.BASE_SPEED
119            speed_left = self.BASE_SPEED / ratio
120
121        self.left_motor.run_angle(speed_left if left_angle >= 0 else -speed_left, abs_left, wait=False, then=then)
122        self.right_motor.run_angle(speed_right if right_angle >= 0 else -speed_right, abs_right, then=then)

142    def move_time(self, duration: float, reverse: bool = False, ease_in: bool = False, ease_out: bool = False, polling_rate: int = 10, then: str = "HOLD", correction: bool = True):
143        """
144        Moves the robot forward or backward for a specific time with optional easing and correction.
145
146        Parameters:
147        - duration: float, movement time in milliseconds.
148        - reverse: bool, move backward if True.
149        - ease_in: bool, gradually increase speed at start.
150        - ease_out: bool, gradually decrease speed at end.
151        - polling_rate: int, time in ms between control updates.
152        - then: str, one of "HOLD", "STOP", or "BRAKE" after movement ends.
153        - correction: bool, enable motor angle correction to keep a straight path.
154        """
155        k = 0.5  # Correction strength
156
157        # Determine easing duration
158        if ease_in and ease_out:
159            ease_duration = min(400, duration / 2)
160        elif ease_in or ease_out:
161            ease_duration = min(400, duration)
162        else:
163            ease_duration = 0
164
165        ease_factor = 0 if ease_in else 1
166
167        start = self.watch.time()
168        delta_time = self.watch.time() - start
169
170        self.left_motor.reset_angle(0)
171        self.right_motor.reset_angle(0)
172
173        while delta_time < duration:
174            # Adjust easing factor
175            if ease_in and delta_time < ease_duration:
176                ease_factor = min(1, delta_time / ease_duration)
177            elif ease_out and duration - delta_time < ease_duration:
178                ease_factor = max(0, (duration - delta_time) / ease_duration)
179
180            # Apply correction if enabled
181            if correction:
182                error = self.right_motor.angle() - self.left_motor.angle()
183                correction_val = k * error
184            else:
185                correction_val = 0
186
187            speed_left = (self.BASE_SPEED - correction_val) * ease_factor
188            speed_right = (self.BASE_SPEED + correction_val) * ease_factor
189
190            if reverse:
191                self.left_motor.run(-speed_left)
192                self.right_motor.run(-speed_right)
193            else:
194                self.left_motor.run(speed_left)
195                self.right_motor.run(speed_right)
196
197            wait(polling_rate)
198            delta_time = self.watch.time() - start
199
200        # Stop behavior
201        if then == "HOLD":
202            self.left_motor.hold()
203            self.right_motor.hold()
204        elif then == "STOP":
205            self.left_motor.stop()
206            self.right_motor.stop()
207        elif then == "BRAKE":
208            self.left_motor.brake()
209            self.right_motor.brake()
